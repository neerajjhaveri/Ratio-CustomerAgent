EXAMPLE (Simplified) QUERY:
WITH base AS (
  SELECT fo.IncidentId,
         fo.Severity,
         fo.IsQCO,
         fo.TTM
  FROM Fact_Outage fo
  WHERE fo.IncidentId IN ({incident_id_list})
),
linkage AS (
  SELECT
    COALESCE(ManualLinkedIncidentId, MLLinkedIncidentId) AS IncidentId,
    CASE WHEN IsCritSit = 1 THEN CaseNumber END AS CritSitCaseNumber,
    CASE WHEN IsCritSit = 1 AND ManualLinkedIncidentId IS NOT NULL THEN CaseNumber END AS ManualCase,
    CASE WHEN IsCritSit = 1 AND MLLinkedIncidentId IS NOT NULL THEN CaseNumber END AS MLCase
  FROM Factless_SupportCaseToIncidentMap
  WHERE (ManualLinkedIncidentId IN ({incident_id_list})
         OR MLLinkedIncidentId IN ({incident_id_list}))
)
SELECT
  b.IncidentId,
  b.Severity,
  b.IsQCO,
  b.TTM,
  COUNT(DISTINCT l.CritSitCaseNumber) AS TotalCritSits,
  COUNT(DISTINCT l.MLCase)            AS MLLinkedCritSits,
  COUNT(DISTINCT l.ManualCase)        AS ManualLinkedCritSits
FROM base b
LEFT JOIN linkage l ON l.IncidentId = b.IncidentId
GROUP BY b.IncidentId, b.Severity, b.IsQCO, b.TTM
ORDER BY b.IncidentId;
