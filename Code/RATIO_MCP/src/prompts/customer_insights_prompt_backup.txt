  AIRO_Customer = SUM(ImpactedResources) / NULLIF(MAX(ActiveResourceCustomerCount),0) GROUP BY IncidentId, Offering, CustomerName

  AIROD_Customer = (SUM(ImpactedResources) * SUM(ImpactDuration)) / NULLIF(MAX(ActiveResourceCustomerCount),0) GROUP BY IncidentId, Offering, CustomerName

CRITSIT METRICS (DOMAIN -> "Outage + SupportCases")
TotalCritSits          = COUNT(DISTINCT CaseNumber) WHERE IsCritSit = 1 linked to IncidentId (manual OR ML)
ManualLinkedCritSits   = COUNT(DISTINCT CaseNumber) WHERE IsCritSit = 1 AND ManualLinkedIncidentId = IncidentId
MLLinkedCritSits       = COUNT(DISTINCT CaseNumber) WHERE IsCritSit = 1 AND MLLinkedIncidentId = IncidentId
FirstCritSitCreatedOn  = MIN(CreatedOn) GROUP BY IncidentId
FirstCustomerCritSitCreatedOn  = MIN(CreatedOn) GROUP BY IncidentId, CustomerName
%CustomerCommsAfterCritSit = (COUNT(CritSit Raising Customers) where CreatedOn is < FirstCommTimeServiceCustomer) / COUNT(Total CritSit Raising Customers)
%CritSitCustomerWithNoComms = (COUNT(CritSit Raising Customers) where Customer is missing from Fact_CustomerTTN) /COUNT(Total CritSit Raising Customers)

CUSTOMER COMMUNICATION METRICS (DOMAIN -> "Outage + OutageCommunication")
Per (IncidentId, ServiceName):
  FirstCommTimeService   = MIN(FirstCommTimeService)
  ServiceTTNInMinutes    = MIN(ServiceTTNInMinutes)
per (IncidentId, CustomerName)
  FirstCommTimeServiceCustomer   = MIN(FirstCommTimeServiceCustomer)
  ServiceCustomerTTNInMinutes    = MIN(ServiceCustomerTTNInMinutes)

DERIVED FLAGS
IsQEICustomer = CASE WHEN CustomerName IN (SELECT CustomerName FROM QEIS500_Customers) THEN 1 ELSE 0 END

CONTRIBUTION %
AIRO_Contribution = SUM(ImpactedResources for customer) / NULLIF(SUM(ImpactedResources for all customers in same IncidentId & basis),0)
