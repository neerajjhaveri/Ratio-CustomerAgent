SAMPLE REGION QUERY
SELECT
  IncidentId,
  Offering,
  Region,
  CASE WHEN ComputationSource IS NOT NULL THEN 'ACM' ELSE 'SLI' END AS ComputationSource,
  MAX(AR_Region) AS ActiveResourceRegionCount,
  SUM(ImpactedResource) AS ImpactedResourceCount,
  CAST(ROUND(1.0 * SUM(ImpactedResource) / NULLIF(MAX(AR_Region),0), 4) AS DECIMAL(12,4)) AS AIRO_Region
FROM dbo.OfferingOutageAiro
WHERE IncidentId = @IncidentId
GROUP BY IncidentId, Offering, Region
ORDER BY AIRO_Region DESC;

SAMPLE CUSTOMER QUERY
SELECT
  IncidentId,
  Offering,
  CustomerName,
  CASE WHEN ComputationSource IS NOT NULL THEN 'ACM' ELSE 'SLI' END AS ComputationSource,
  MAX(AR_Customer) AS ActiveResourceCustomerCount,
  SUM(ImpactedResource) AS ImpactedResourceCount,
  CAST(ROUND(1.0 * SUM(ImpactedResource) / NULLIF(MAX(AR_Customer),0), 4) AS DECIMAL(12,4)) AS AIRO_Customer
FROM dbo.OfferingOutageAiro
WHERE IncidentId = @IncidentId
GROUP BY IncidentId, Offering, CustomerName
ORDER BY AIRO_Customer DESC;
